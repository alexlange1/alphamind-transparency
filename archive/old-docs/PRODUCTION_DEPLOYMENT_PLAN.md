```bash
#!/bin/bash
# deploy_production.sh

set -e

echo "üöÄ TAO20 Production Deployment"
echo "=============================="

# Configuration
NETWORK=${1:-"bevm-testnet"}
PRIVATE_KEY=${2:-$BEVM_PRIVATE_KEY}
VERIFY=${3:-"false"}

if [ -z "$PRIVATE_KEY" ]; then
    echo "‚ùå Error: Private key not provided"
    exit 1
fi

# Network configurations
case $NETWORK in
    "bevm-testnet")
        RPC_URL="https://testnet-rpc.bevm.io"
        CHAIN_ID="1501"
        VERIFY_FLAG="--verify"
        ;;
    "bevm-mainnet")
        RPC_URL="https://rpc-mainnet-1.bevm.io"
        CHAIN_ID="11501"
        VERIFY_FLAG=""
        ;;
    "private")
        RPC_URL="http://localhost:8545"
        CHAIN_ID="11502"
        VERIFY_FLAG=""
        ;;
    *)
        echo "‚ùå Error: Unknown network $NETWORK"
        exit 1
        ;;
esac

echo "üì° Network: $NETWORK"
echo "üîó RPC: $RPC_URL"
echo "üÜî Chain ID: $CHAIN_ID"

# Pre-deployment checks
echo ""
echo "üîç Pre-deployment Validation"
echo "----------------------------"

# Check RPC connectivity
if ! curl -s -X POST -H "Content-Type: application/json" \
    --data '{"jsonrpc":"2.0","method":"eth_blockNumber","params":[],"id":1}' \
    $RPC_URL > /dev/null; then
    echo "‚ùå Error: Cannot connect to RPC"
    exit 1
fi
echo "‚úÖ RPC connectivity confirmed"

# Check account balance
BALANCE=$(cast balance --rpc-url $RPC_URL $(cast wallet address $PRIVATE_KEY))
BALANCE_ETH=$(cast to-unit $BALANCE ether)
echo "üí∞ Deployer balance: $BALANCE_ETH ETH"

if (( $(echo "$BALANCE_ETH < 0.1" | bc -l) )); then
    echo "‚ö†Ô∏è  Warning: Low balance for deployment"
fi

# Compile contracts
echo ""
echo "üî® Contract Compilation"
echo "----------------------"
cd contracts
forge build
echo "‚úÖ Contracts compiled successfully"

# Run tests
echo ""
echo "üß™ Final Testing"
echo "---------------"
forge test --match-contract TAO20SimpleFlowTest
echo "‚úÖ Tests passed"

# Deploy contracts
echo ""
echo "üöÄ Contract Deployment"
echo "---------------------"
forge script script/DeployProduction.s.sol \
    --rpc-url $RPC_URL \
    --private-key $PRIVATE_KEY \
    --broadcast \
    $VERIFY_FLAG \
    --legacy

if [ $? -eq 0 ]; then
    echo "‚úÖ Deployment successful!"
    echo ""
    echo "üìÑ Deployment Details:"
    echo "----------------------"
    
    # Extract contract addresses from deployment
    DEPLOYMENT_FILE="broadcast/DeployProduction.s.sol/$CHAIN_ID/run-latest.json"
    if [ -f "$DEPLOYMENT_FILE" ]; then
        echo "üìÅ Deployment file: $DEPLOYMENT_FILE"
        # Parse and display contract addresses
        echo "üèóÔ∏è  Contract addresses saved to deployment file"
    fi
    
    echo ""
    echo "üéâ Deployment Complete!"
    echo "Ready for testing and validation"
else
    echo "‚ùå Deployment failed"
    exit 1
fi
```

### **üìä Post-Deployment Validation**

```python
#!/usr/bin/env python3
# validate_deployment.py

import asyncio
import json
import sys
from web3 import Web3
from eth_account import Account

async def validate_deployment(network_config):
    """Validate deployed contracts"""
    
    print("üîç Post-Deployment Validation")
    print("=" * 40)
    
    # Initialize Web3
    w3 = Web3(Web3.HTTPProvider(network_config['rpc_url']))
    if not w3.is_connected():
        print("‚ùå Cannot connect to network")
        return False
    
    print(f"‚úÖ Connected to {network_config['name']}")
    
    # Load deployment addresses
    with open(network_config['deployment_file'], 'r') as f:
        deployment = json.load(f)
    
    # Extract contract addresses
    contracts = {}
    for tx in deployment['transactions']:
        if 'contractAddress' in tx:
            # Map contract addresses
            contracts[tx['contractName']] = tx['contractAddress']
    
    print(f"üìÑ Found {len(contracts)} deployed contracts")
    
    # Validation tests
    tests_passed = 0
    total_tests = 0
    
    # Test 1: Contract code verification
    total_tests += 1
    print("\nüß™ Test 1: Contract Code Verification")
    try:
        for name, address in contracts.items():
            code = w3.eth.get_code(address)
            if len(code) > 2:  # More than "0x"
                print(f"  ‚úÖ {name}: Code deployed")
            else:
                print(f"  ‚ùå {name}: No code found")
                raise Exception(f"No code at {address}")
        tests_passed += 1
        print("‚úÖ All contracts have code deployed")
    except Exception as e:
        print(f"‚ùå Contract code verification failed: {e}")
    
    # Test 2: Basic function calls
    total_tests += 1
    print("\nüß™ Test 2: Basic Function Calls")
    try:
        # Test NAV calculator if available
        if 'NAVCalculator' in contracts:
            nav_addr = contracts['NAVCalculator']
            # Simple call to getCurrentNAV
            nav_abi = [{"inputs":[],"name":"getCurrentNAV","outputs":[{"name":"","type":"uint256"}],"stateMutability":"view","type":"function"}]
            nav_contract = w3.eth.contract(address=nav_addr, abi=nav_abi)
            nav = nav_contract.functions.getCurrentNAV().call()
            print(f"  ‚úÖ NAV Calculator: {nav}")
        
        tests_passed += 1
        print("‚úÖ Basic function calls working")
    except Exception as e:
        print(f"‚ùå Basic function calls failed: {e}")
    
    # Test 3: Gas cost analysis
    total_tests += 1
    print("\nüß™ Test 3: Gas Cost Analysis")
    try:
        gas_prices = {}
        for name, address in contracts.items():
            # Estimate gas for simple calls
            gas_prices[name] = "Analysis needed"
        
        tests_passed += 1
        print("‚úÖ Gas analysis available")
    except Exception as e:
        print(f"‚ùå Gas analysis failed: {e}")
    
    # Results
    print(f"\nüìä Validation Results: {tests_passed}/{total_tests} tests passed")
    
    if tests_passed == total_tests:
        print("üéâ Deployment validation successful!")
        return True
    else:
        print("‚ö†Ô∏è  Some validation tests failed")
        return False

if __name__ == "__main__":
    network = sys.argv[1] if len(sys.argv) > 1 else "bevm-testnet"
    
    configs = {
        "bevm-testnet": {
            "name": "BEVM Testnet",
            "rpc_url": "https://testnet-rpc.bevm.io",
            "deployment_file": "broadcast/DeployProduction.s.sol/1501/run-latest.json"
        },
        "bevm-mainnet": {
            "name": "BEVM Mainnet", 
            "rpc_url": "https://rpc-mainnet-1.bevm.io",
            "deployment_file": "broadcast/DeployProduction.s.sol/11501/run-latest.json"
        }
    }
    
    if network not in configs:
        print(f"‚ùå Unknown network: {network}")
        sys.exit(1)
    
    success = asyncio.run(validate_deployment(configs[network]))
    sys.exit(0 if success else 1)
```
